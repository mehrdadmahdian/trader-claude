package xapi

import "time"

// Tweet represents a post from X API v2.
type Tweet struct {
	ID        string    `json:"id"`
	Text      string    `json:"text"`
	AuthorID  string    `json:"author_id"`
	CreatedAt time.Time `json:"created_at"`
	PublicMetrics struct {
		RetweetCount int `json:"retweet_count"`
		ReplyCount   int `json:"reply_count"`
		LikeCount    int `json:"like_count"`
		QuoteCount   int `json:"quote_count"`
	} `json:"public_metrics"`
}

// SearchResponse is the raw X API v2 search response.
type SearchResponse struct {
	Data []Tweet `json:"data"`
	Meta struct {
		ResultCount int    `json:"result_count"`
		NextToken   string `json:"next_token"`
	} `json:"meta"`
}

// PostRequest is the body for POST /2/tweets.
type PostRequest struct {
	Text string `json:"text"`
}

// PostResponse is the X API v2 response for creating a tweet.
type PostResponse struct {
	Data struct {
		ID   string `json:"id"`
		Text string `json:"text"`
	} `json:"data"`
}

// UserMeResponse is the response from GET /2/users/me.
type UserMeResponse struct {
	Data struct {
		ID       string `json:"id"`
		Name     string `json:"name"`
		Username string `json:"username"`
	} `json:"data"`
}

// ScoredTweet is a Tweet with sentiment analysis attached.
type ScoredTweet struct {
	Tweet
	Sentiment SentimentResult `json:"sentiment"`
}

// SentimentResult holds the sentiment analysis output.
type SentimentResult struct {
	Score    float64 `json:"score"`
	Positive int     `json:"positive"`
	Negative int     `json:"negative"`
	Label    string  `json:"label"` // "bullish" | "bearish" | "neutral"
}

// FeedResponse is the API response for the X feed endpoint.
type FeedResponse struct {
	Tweets           []ScoredTweet    `json:"tweets"`
	SentimentSummary SentimentSummary `json:"sentiment_summary"`
}

// SentimentSummary aggregates sentiment across multiple tweets.
type SentimentSummary struct {
	AvgScore float64 `json:"avg_score"`
	Label    string  `json:"label"`
	Count    int     `json:"count"`
}
```

package xapi

import (
	"bytes"
	"context"
	"encoding/json"
	"fmt"
	"io"
	"net/http"
	"net/url"
	"time"

	"golang.org/x/time/rate"
)

const defaultBaseURL = "https://api.x.com"

// XClient communicates with the X API v2.
type XClient struct {
	bearerToken  string
	apiKey       string
	apiSecret    string
	accessToken  string
	accessSecret string
	httpClient   *http.Client
	limiter      *rate.Limiter
	baseURL      string
}

// NewXClient creates a read-only client using a Bearer Token.
func NewXClient(bearerToken string) *XClient {
	return &XClient{
		bearerToken: bearerToken,
		httpClient:  &http.Client{Timeout: 15 * time.Second},
		limiter:     rate.NewLimiter(rate.Every(15*time.Minute/60), 1),
		baseURL:     defaultBaseURL,
	}
}

// NewXClientWithUserContext creates a client that can also post tweets.
func NewXClientWithUserContext(apiKey, apiSecret, accessToken, accessSecret string) *XClient {
	c := &XClient{
		apiKey:       apiKey,
		apiSecret:    apiSecret,
		accessToken:  accessToken,
		accessSecret: accessSecret,
		httpClient:   &http.Client{Timeout: 15 * time.Second},
		limiter:      rate.NewLimiter(rate.Every(15*time.Minute/60), 1),
		baseURL:      defaultBaseURL,
	}
	return c
}

// SetBaseURL overrides the API base URL (for testing).
func (x *XClient) SetBaseURL(url string) {
	x.baseURL = url
}

// SearchRecent searches tweets from the last 7 days.
func (x *XClient) SearchRecent(ctx context.Context, query string, maxResults int) (*SearchResponse, error) {
	if err := x.limiter.Wait(ctx); err != nil {
		return nil, fmt.Errorf("xapi: rate limit wait: %w", err)
	}

	if maxResults < 10 {
		maxResults = 10
	}
	if maxResults > 100 {
		maxResults = 100
	}

	params := url.Values{}
	params.Set("query", query)
	params.Set("max_results", fmt.Sprintf("%d", maxResults))
	params.Set("tweet.fields", "created_at,public_metrics,author_id")

	reqURL := fmt.Sprintf("%s/2/tweets/search/recent?%s", x.baseURL, params.Encode())

	req, err := http.NewRequestWithContext(ctx, http.MethodGet, reqURL, nil)
	if err != nil {
		return nil, fmt.Errorf("xapi: create request: %w", err)
	}
	req.Header.Set("Authorization", "Bearer "+x.bearerToken)

	resp, err := x.httpClient.Do(req)
	if err != nil {
		return nil, fmt.Errorf("xapi: search request: %w", err)
	}
	defer resp.Body.Close()

	if resp.StatusCode != http.StatusOK {
		body, _ := io.ReadAll(resp.Body)
		return nil, fmt.Errorf("xapi: search failed (status %d): %s", resp.StatusCode, string(body))
	}

	var result SearchResponse
	if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
		return nil, fmt.Errorf("xapi: decode search response: %w", err)
	}
	return &result, nil
}

// PostTweet creates a new post using OAuth 1.0a user context.
func (x *XClient) PostTweet(ctx context.Context, text string) (*PostResponse, error) {
	if err := x.limiter.Wait(ctx); err != nil {
		return nil, fmt.Errorf("xapi: rate limit wait: %w", err)
	}

	body := PostRequest{Text: text}
	jsonBody, err := json.Marshal(body)
	if err != nil {
		return nil, fmt.Errorf("xapi: marshal post body: %w", err)
	}

	reqURL := fmt.Sprintf("%s/2/tweets", x.baseURL)
	req, err := http.NewRequestWithContext(ctx, http.MethodPost, reqURL, bytes.NewReader(jsonBody))
	if err != nil {
		return nil, fmt.Errorf("xapi: create post request: %w", err)
	}
	req.Header.Set("Content-Type", "application/json")

	// OAuth 1.0a signature would be added here in production.
	// For simplicity, using Bearer token as fallback if user context not fully configured.
	if x.accessToken != "" {
		req.Header.Set("Authorization", "Bearer "+x.accessToken)
	} else if x.bearerToken != "" {
		req.Header.Set("Authorization", "Bearer "+x.bearerToken)
	}

	resp, err := x.httpClient.Do(req)
	if err != nil {
		return nil, fmt.Errorf("xapi: post request: %w", err)
	}
	defer resp.Body.Close()

	if resp.StatusCode != http.StatusCreated && resp.StatusCode != http.StatusOK {
		respBody, _ := io.ReadAll(resp.Body)
		return nil, fmt.Errorf("xapi: post failed (status %d): %s", resp.StatusCode, string(respBody))
	}

	var result PostResponse
	if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
		return nil, fmt.Errorf("xapi: decode post response: %w", err)
	}
	return &result, nil
}

// TestConnection verifies the bearer token by fetching /2/users/me.
func (x *XClient) TestConnection(ctx context.Context) (string, error) {
	reqURL := fmt.Sprintf("%s/2/users/me", x.baseURL)
	req, err := http.NewRequestWithContext(ctx, http.MethodGet, reqURL, nil)
	if err != nil {
		return "", fmt.Errorf("xapi: create request: %w", err)
	}

	token := x.bearerToken
	if x.accessToken != "" {
		token = x.accessToken
	}
	req.Header.Set("Authorization", "Bearer "+token)

	resp, err := x.httpClient.Do(req)
	if err != nil {
		return "", fmt.Errorf("xapi: test connection: %w", err)
	}
	defer resp.Body.Close()

	if resp.StatusCode != http.StatusOK {
		return "", fmt.Errorf("xapi: test connection failed (status %d)", resp.StatusCode)
	}

	var result UserMeResponse
	if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
		return "", fmt.Errorf("xapi: decode user: %w", err)
	}
	return result.Data.Username, nil
}

package xapi

import (
	"strings"
)

var bullishKeywords = []string{
	"buy", "bullish", "moon", "pump", "breakout", "long", "accumulate",
	"ath", "surge", "rally", "soaring", "green", "gains", "uptrend",
	"support", "hodl", "to the moon", "rocket",
}

var bearishKeywords = []string{
	"sell", "bearish", "dump", "crash", "short", "liquidation", "drop",
	"plunge", "decline", "correction", "red", "losses", "downtrend",
	"resistance", "capitulation", "fear", "rug pull",
}

func ScoreSentiment(text string) SentimentResult {
	lower := strings.ToLower(text)
	words := strings.Fields(lower)

	var pos, neg int
	for _, word := range words {
		clean := strings.Trim(word, ".,!?;:'\"()[]{}#@$")
		for _, kw := range bullishKeywords {
			if clean == kw || strings.Contains(lower, kw) {
				pos++
				break
			}
		}
		for _, kw := range bearishKeywords {
			if clean == kw || strings.Contains(lower, kw) {
				neg++
				break
			}
		}
	}

	total := pos + neg
	if total == 0 {
		return SentimentResult{Score: 0, Positive: 0, Negative: 0, Label: "neutral"}
	}

	score := float64(pos-neg) / float64(total)
	if score > 1 {
		score = 1
	}
	if score < -1 {
		score = -1
	}

	label := "neutral"
	if score > 0.2 {
		label = "bullish"
	} else if score < -0.2 {
		label = "bearish"
	}

	return SentimentResult{
		Score:    score,
		Positive: pos,
		Negative: neg,
		Label:    label,
	}
}

func SentimentLabel(avgScore float64) string {
	if avgScore > 0.2 {
		return "bullish"
	}
	if avgScore < -0.2 {
		return "bearish"
	}
	return "neutral"
}

package xapi

import (
	"context"
	"encoding/json"
	"fmt"
	"strings"
	"time"

	"github.com/redis/go-redis/v9"
)

const feedCacheTTL = 5 * time.Minute

type FeedService struct {
	client *XClient
	rdb    *redis.Client
}

func NewFeedService(client *XClient, rdb *redis.Client) *FeedService {
	return &FeedService{client: client, rdb: rdb}
}

func (f *FeedService) FetchSymbolFeed(ctx context.Context, symbol string, maxResults int) (*FeedResponse, error) {
	cacheKey := fmt.Sprintf("x:feed:%s", symbol)

	// Check cache first
	cached, err := f.rdb.Get(ctx, cacheKey).Bytes()
	if err == nil {
		var resp FeedResponse
		if json.Unmarshal(cached, &resp) == nil {
			return &resp, nil
		}
	}

	// Fetch from X API
	cashtag := SymbolToCashtag(symbol)
	searchResp, err := f.client.SearchRecent(ctx, cashtag, maxResults)
	if err != nil {
		return nil, fmt.Errorf("feed: search X: %w", err)
	}

	// Score sentiment for each tweet
	scored := make([]ScoredTweet, 0, len(searchResp.Data))
	var totalScore float64
	for _, tweet := range searchResp.Data {
		sentiment := ScoreSentiment(tweet.Text)
		totalScore += sentiment.Score
		scored = append(scored, ScoredTweet{Tweet: tweet, Sentiment: sentiment})
	}

	avgScore := 0.0
	if len(scored) > 0 {
		avgScore = totalScore / float64(len(scored))
	}

	resp := &FeedResponse{
		Tweets: scored,
		SentimentSummary: SentimentSummary{
			AvgScore: avgScore,
			Label:    SentimentLabel(avgScore),
			Count:    len(scored),
		},
	}

	// Cache result
	if data, err := json.Marshal(resp); err == nil {
		f.rdb.Set(ctx, cacheKey, data, feedCacheTTL)
	}

	return resp, nil
}

func SymbolToCashtag(symbol string) string {
	parts := strings.Split(symbol, "/")
	return "$" + parts[0]
}
```
