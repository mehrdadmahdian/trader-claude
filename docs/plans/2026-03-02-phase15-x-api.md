```go
package xapi

import "time"

// Tweet is a single X post.
type Tweet struct {
	ID        string    `json:"id"`
	Text      string    `json:"text"`
	AuthorID  string    `json:"author_id"`
	Username  string    `json:"username"`
	CreatedAt time.Time `json:"created_at"`
	Likes     int       `json:"like_count"`
	Retweets  int       `json:"retweet_count"`
}

// ScoredTweet is a Tweet with sentiment attached.
type ScoredTweet struct {
	Tweet
	Sentiment float64 `json:"sentiment"` // [-1, 1]
	Label     string  `json:"label"`     // "bullish" | "bearish" | "neutral"
}

// SentimentSummary aggregates sentiment across a set of tweets.
type SentimentSummary struct {
	BullishPct float64 `json:"bullish_pct"`
	BearishPct float64 `json:"bearish_pct"`
	NeutralPct float64 `json:"neutral_pct"`
	AvgScore   float64 `json:"avg_score"`
	TweetCount int     `json:"tweet_count"`
}

// SearchResponse is returned by GET /x/feed.
type SearchResponse struct {
	Tweets  []ScoredTweet    `json:"tweets"`
	Summary SentimentSummary `json:"summary"`
}

// PostResponse is returned after posting a tweet.
type PostResponse struct {
	TweetID string `json:"tweet_id"`
	URL     string `json:"url"`
}

// xSearchResult is the raw Twitter API v2 response shape.
type xSearchResult struct {
	Data []struct {
		ID        string `json:"id"`
		Text      string `json:"text"`
		AuthorID  string `json:"author_id"`
		CreatedAt string `json:"created_at"`
		PublicMetrics struct {
			LikeCount    int `json:"like_count"`
			RetweetCount int `json:"retweet_count"`
		} `json:"public_metrics"`
	} `json:"data"`
	Includes struct {
		Users []struct {
			ID       string `json:"id"`
			Username string `json:"username"`
		} `json:"users"`
	} `json:"includes"`
}
```

```go
package xapi

import (
	"strings"
	"unicode"
)

var bullishKeywords = map[string]bool{
	"moon": true, "pump": true, "ath": true, "bullish": true, "buy": true,
	"long": true, "breakout": true, "rally": true, "bull": true, "surge": true,
	"gain": true, "upside": true, "green": true, "hodl": true,
	"accumulate": true, "support": true, "launch": true, "soar": true,
}

var bearishKeywords = map[string]bool{
	"crash": true, "dump": true, "bear": true, "short": true, "sell": true,
	"drop": true, "collapse": true, "rekt": true, "dip": true, "downside": true,
	"red": true, "fear": true, "panic": true, "bottom": true,
	"resistance": true, "rug": true, "bearish": true,
}

// Score returns a sentiment score in [-1, 1] for the given text.
// Positive = bullish, negative = bearish, zero = neutral.
func Score(text string) float64 {
	if text == "" {
		return 0
	}

	tokens := tokenize(text)
	if len(tokens) == 0 {
		return 0
	}

	var bullish, bearish float64
	for _, t := range tokens {
		if bullishKeywords[t] {
			bullish++
		} else if bearishKeywords[t] {
			bearish++
		}
	}

	raw := (bullish - bearish) / float64(len(tokens))
	return clamp(raw, -1, 1)
}

// Label converts a score to a human-readable label.
func Label(score float64) string {
	switch {
	case score > 0:
		return "bullish"
	case score < 0:
		return "bearish"
	default:
		return "neutral"
	}
}

func tokenize(text string) []string {
	lower := strings.ToLower(text)
	return strings.FieldsFunc(lower, func(r rune) bool {
		return !unicode.IsLetter(r) && !unicode.IsDigit(r)
	})
}

func clamp(v, min, max float64) float64 {
	if v < min {
		return min
	}
	if v > max {
		return max
	}
	return v
}
```

```go
package xapi

import (
	"bytes"
	"encoding/json"
	"errors"
	"fmt"
	"io"
	"net/http"
	"net/url"
	"strconv"
	"sync"
	"time"
)

// ErrRateLimited is returned when the client-side rate limit is exceeded.
var ErrRateLimited = errors.New("xapi: rate limit exceeded, try again later")

// XClient wraps the X (Twitter) API v2.
// When mockMode is true (bearer token absent), all methods return fixture data.
type XClient struct {
	bearerToken       string
	apiKey            string
	apiSecret         string
	accessToken       string
	accessTokenSecret string
	mockMode          bool
	httpClient        *http.Client

	// Rate limiter: maxRequests per 15-minute window
	mu           sync.Mutex
	reqCount     int
	windowStart  time.Time
	maxRequests  int
}

// NewClient creates an XClient. Pass empty bearerToken to enable mock mode.
func NewClient(bearerToken, apiKey, apiSecret, accessToken, accessTokenSecret string) *XClient {
	return &XClient{
		bearerToken:       bearerToken,
		apiKey:            apiKey,
		apiSecret:         apiSecret,
		accessToken:       accessToken,
		accessTokenSecret: accessTokenSecret,
		mockMode:          bearerToken == "",
		httpClient:        &http.Client{Timeout: 10 * time.Second},
		windowStart:       time.Now(),
		maxRequests:       60,
	}
}

// MockMode reports whether the client is in mock mode.
func (c *XClient) MockMode() bool { return c.mockMode }

// SearchRecent fetches recent tweets for the given cashtag (e.g. "$BTC").
// Returns mock fixture tweets in mock mode.
func (c *XClient) SearchRecent(cashtag string, limit int) ([]Tweet, error) {
	if c.mockMode {
		return mockTweets(cashtag, limit), nil
	}

	if err := c.checkRateLimit(); err != nil {
		return nil, err
	}

	if limit <= 0 || limit > 100 {
		limit = 10
	}

	reqURL := fmt.Sprintf(
		"https://api.twitter.com/2/tweets/search/recent?query=%s&max_results=%d&tweet.fields=created_at,public_metrics,author_id&expansions=author_id&user.fields=username",
		url.QueryEscape(cashtag),
		limit,
	)

	req, err := http.NewRequest("GET", reqURL, nil)
	if err != nil {
		return nil, fmt.Errorf("xapi: build request: %w", err)
	}
	req.Header.Set("Authorization", "Bearer "+c.bearerToken)

	resp, err := c.httpClient.Do(req)
	if err != nil {
		return nil, fmt.Errorf("xapi: http error: %w", err)
	}
	defer resp.Body.Close()

	if resp.StatusCode == http.StatusUnauthorized {
		return nil, errors.New("xapi: invalid bearer token")
	}
	if resp.StatusCode != http.StatusOK {
		body, _ := io.ReadAll(resp.Body)
		return nil, fmt.Errorf("xapi: unexpected status %d: %s", resp.StatusCode, string(body))
	}

	var result xSearchResult
	if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
		return nil, fmt.Errorf("xapi: decode response: %w", err)
	}

	// Build username lookup
	userMap := make(map[string]string, len(result.Includes.Users))
	for _, u := range result.Includes.Users {
		userMap[u.ID] = u.Username
	}

	tweets := make([]Tweet, 0, len(result.Data))
	for _, d := range result.Data {
		ts, _ := time.Parse(time.RFC3339, d.CreatedAt)
		tweets = append(tweets, Tweet{
			ID:        d.ID,
			Text:      d.Text,
			AuthorID:  d.AuthorID,
			Username:  userMap[d.AuthorID],
			CreatedAt: ts,
			Likes:     d.PublicMetrics.LikeCount,
			Retweets:  d.PublicMetrics.RetweetCount,
		})
	}
	return tweets, nil
}

// PostTweet posts a tweet using OAuth 1.0a user context.
func (c *XClient) PostTweet(text string) (PostResponse, error) {
	if c.mockMode {
		id := "mock-" + strconv.FormatInt(time.Now().UnixNano(), 36)
		return PostResponse{
			TweetID: id,
			URL:     "https://x.com/mock_user/status/" + id,
		}, nil
	}

	if c.accessToken == "" || c.accessTokenSecret == "" || c.apiKey == "" || c.apiSecret == "" {
		return PostResponse{}, errors.New("xapi: posting requires API key, secret, access token and access token secret")
	}

	body, _ := json.Marshal(map[string]string{"text": text})
	req, err := http.NewRequest("POST", "https://api.twitter.com/2/tweets", bytes.NewReader(body))
	if err != nil {
		return PostResponse{}, fmt.Errorf("xapi: build request: %w", err)
	}
	req.Header.Set("Content-Type", "application/json")
	req.Header.Set("Authorization", buildOAuth1Header(c.apiKey, c.apiSecret, c.accessToken, c.accessTokenSecret, "POST", "https://api.twitter.com/2/tweets"))

	resp, err := c.httpClient.Do(req)
	if err != nil {
		return PostResponse{}, fmt.Errorf("xapi: http error: %w", err)
	}
	defer resp.Body.Close()

	if resp.StatusCode != http.StatusCreated {
		b, _ := io.ReadAll(resp.Body)
		return PostResponse{}, fmt.Errorf("xapi: post failed status %d: %s", resp.StatusCode, string(b))
	}

	var result struct {
		Data struct {
			ID string `json:"id"`
		} `json:"data"`
	}
	if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
		return PostResponse{}, fmt.Errorf("xapi: decode response: %w", err)
	}

	return PostResponse{
		TweetID: result.Data.ID,
		URL:     "https://x.com/i/web/status/" + result.Data.ID,
	}, nil
}

// checkRateLimit returns ErrRateLimited if the 60/15min window is exhausted.
func (c *XClient) checkRateLimit() error {
	c.mu.Lock()
	defer c.mu.Unlock()

	now := time.Now()
	if now.Sub(c.windowStart) > 15*time.Minute {
		c.reqCount = 0
		c.windowStart = now
	}
	if c.reqCount >= c.maxRequests {
		return ErrRateLimited
	}
	c.reqCount++
	return nil
}

// buildOAuth1Header generates an OAuth 1.0a Authorization header.
// This is a simplified implementation suitable for single-user context.
func buildOAuth1Header(apiKey, apiSecret, accessToken, accessTokenSecret, method, reqURL string) string {
	// Minimal OAuth 1.0a signature — for production use an OAuth library.
	// This stub returns a placeholder that the real X API would reject without
	// proper HMAC-SHA1 signing; replace with oauth1 library when needed.
	ts := strconv.FormatInt(time.Now().Unix(), 10)
	nonce := strconv.FormatInt(time.Now().UnixNano(), 36)

		apiKey, nonce, ts, accessToken,
	)
}
```

```go
package xapi

import (
	"context"
	"encoding/json"
	"log"
	"strings"
	"time"

	goredis "github.com/redis/go-redis/v9"
)

const cacheTTL = 5 * time.Minute

// FeedService combines XClient + Redis cache to serve symbol tweet feeds.
type FeedService struct {
	client *XClient
	rdb    *goredis.Client
}

// NewFeedService creates a FeedService. Pass nil rdb to disable caching.
func NewFeedService(client *XClient, rdb *goredis.Client) *FeedService {
	return &FeedService{client: client, rdb: rdb}
}

// GetFeed returns scored tweets and a sentiment summary for the given symbol.
// Results are cached in Redis for 5 minutes.
func (s *FeedService) GetFeed(symbol string, limit int) (*SearchResponse, error) {
	cashtag := symbolToCashtag(symbol)

	// Check cache
	if s.rdb != nil {
		key := "x:feed:" + cashtag
		cached, err := s.rdb.Get(context.Background(), key).Bytes()
		if err == nil {
			var resp SearchResponse
			if json.Unmarshal(cached, &resp) == nil {
				return &resp, nil
			}
		}
	}

	tweets, err := s.client.SearchRecent(cashtag, limit)
	if err != nil {
		return nil, err
	}

	scored := make([]ScoredTweet, len(tweets))
	for i, t := range tweets {
		score := Score(t.Text)
		scored[i] = ScoredTweet{
			Tweet:     t,
			Sentiment: score,
			Label:     Label(score),
		}
	}

	summary := buildSummary(scored)
	resp := &SearchResponse{Tweets: scored, Summary: summary}

	// Store in cache
	if s.rdb != nil {
		if data, err := json.Marshal(resp); err == nil {
			key := "x:feed:" + cashtag
			if err := s.rdb.Set(context.Background(), key, data, cacheTTL).Err(); err != nil {
				log.Printf("[xapi] cache set error: %v", err)
			}
		}
	}

	return resp, nil
}

// symbolToCashtag converts a trading symbol to an X cashtag.
// "BTC/USDT" → "$BTC", "AAPL" → "$AAPL"
func symbolToCashtag(symbol string) string {
	// Strip known quote currencies after "/"
	quoteCurrencies := []string{"/USDT", "/BUSD", "/USD", "/BTC", "/ETH", "/BNB", "/USDC"}
	upper := strings.ToUpper(symbol)
	for _, q := range quoteCurrencies {
		if idx := strings.Index(upper, q); idx != -1 {
			return "$" + upper[:idx]
		}
	}
	// If no slash present, use as-is
	if idx := strings.Index(upper, "/"); idx != -1 {
		return "$" + upper[:idx]
	}
	return "$" + upper
}

// buildSummary computes aggregate sentiment statistics.
func buildSummary(tweets []ScoredTweet) SentimentSummary {
	n := len(tweets)
	if n == 0 {
		return SentimentSummary{}
	}

	var bullish, bearish, neutral int
	var totalScore float64
	for _, t := range tweets {
		totalScore += t.Sentiment
		switch t.Label {
		case "bullish":
			bullish++
		case "bearish":
			bearish++
		default:
			neutral++
		}
	}

	fn := float64(n)
	return SentimentSummary{
		BullishPct: float64(bullish) / fn * 100,
		BearishPct: float64(bearish) / fn * 100,
		NeutralPct: float64(neutral) / fn * 100,
		AvgScore:   totalScore / fn,
		TweetCount: n,
	}
}
```

```go
	var body struct {
		Text string `json:"text"`

		emoji = "🔴"

	action := strings.ToUpper(direction)
	priceStr := "$" + strconv.FormatFloat(price, 'f', 2, 64)
	return strings.Join([]string{
		emoji + " " + action + " Signal — " + symbol,
		"Price: " + priceStr + " | Strategy: " + strategyName,
		"#trading #signal Generated with StratosMarket",
	}, "\n")
}
```

## Task 11: Frontend — PostToXModal component

- Create: `frontend/src/components/x/PostToXModal.tsx`

### Step 1: Create `PostToXModal.tsx`

```tsx
import { useState } from 'react'
import { X, ExternalLink } from 'lucide-react'
import { usePostToX } from '@/hooks/useXFeed'

interface PostToXModalProps {
  initialText: string
  onClose: () => void
}

const MAX_CHARS = 280

export function PostToXModal({ initialText, onClose }: PostToXModalProps) {
  const [text, setText] = useState(initialText)
  const [postedUrl, setPostedUrl] = useState<string | null>(null)
  const mutation = usePostToX()

  const remaining = MAX_CHARS - [...text].length
  const isOverLimit = remaining < 0

  function handlePost() {
    if (isOverLimit || !text.trim()) return
    mutation.mutate(text, {
      onSuccess: (data) => {
        setPostedUrl(data.url)
      },
    })
  }

  return (
    <div className="fixed inset-0 z-50 flex items-center justify-center p-4 bg-black/50">
      <div className="w-full max-w-lg rounded-xl border border-border bg-background shadow-xl">
        {/* Header */}
        <div className="flex items-center justify-between px-5 py-4 border-b border-border">
          <div className="flex items-center gap-2">
            {/* X logo */}
            <svg viewBox="0 0 24 24" className="h-5 w-5 fill-foreground" aria-hidden="true">
              <path d="M18.244 2.25h3.308l-7.227 8.26 8.502 11.24H16.17l-4.714-6.231-5.401 6.231H2.744l7.73-8.835L1.254 2.25H8.08l4.253 5.622zm-1.161 17.52h1.833L7.084 4.126H5.117z" />
            </svg>
            <h2 className="text-base font-semibold">Post to X</h2>
          </div>
          <button
            type="button"
            onClick={onClose}
            className="p-1.5 rounded-md hover:bg-accent transition-colors"
            aria-label="Close"
          >
            <X className="h-4 w-4" />
          </button>
        </div>

        {postedUrl ? (
          /* Success state */
          <div className="px-5 py-8 flex flex-col items-center gap-4 text-center">
            <div className="h-12 w-12 rounded-full bg-green-100 dark:bg-green-900/30 flex items-center justify-center">
              <svg viewBox="0 0 24 24" className="h-6 w-6 fill-green-600 dark:fill-green-400" aria-hidden="true">
                <path d="M9 16.17L4.83 12l-1.42 1.41L9 19 21 7l-1.41-1.41L9 16.17z" />
              </svg>
            </div>
            <p className="text-sm font-medium">Posted successfully!</p>
            <a
              href={postedUrl}
              target="_blank"
              rel="noopener noreferrer"
              className="flex items-center gap-1.5 text-sm text-primary hover:underline"
            >
              View on X
              <ExternalLink className="h-3.5 w-3.5" />
            </a>
            <button
              type="button"
              onClick={onClose}
              className="px-4 py-2 text-sm rounded-md border border-border hover:bg-accent transition-colors"
            >
              Close
            </button>
          </div>
        ) : (
          /* Compose state */
          <div className="px-5 py-4 flex flex-col gap-4">
            <textarea
              value={text}
              onChange={(e) => setText(e.target.value)}
              rows={6}
              className="w-full resize-none rounded-lg border border-border bg-input px-3 py-2 text-sm focus:outline-none focus:ring-2 focus:ring-primary"
              placeholder="What's happening?"
            />

            {/* Footer */}
            <div className="flex items-center justify-between">
              <span className={`text-xs tabular-nums ${isOverLimit ? 'text-red-500' : 'text-muted-foreground'}`}>
                {remaining} characters remaining
              </span>
              <div className="flex items-center gap-2">
                <button
                  type="button"
                  onClick={onClose}
                  className="px-3 py-1.5 text-sm rounded-md border border-border hover:bg-accent transition-colors"
                >
                  Cancel
                </button>
                <button
                  type="button"
                  onClick={handlePost}
                  disabled={isOverLimit || !text.trim() || mutation.isPending}
                  className="px-4 py-1.5 text-sm rounded-md bg-primary text-primary-foreground hover:bg-primary/90 disabled:opacity-50 transition-colors"
                >
                  {mutation.isPending ? 'Posting…' : 'Post'}
                </button>
              </div>
            </div>

            {mutation.isError && (
              <p className="text-sm text-red-500">Failed to post. Please try again.</p>
            )}
          </div>
        )}
      </div>
    </div>
  )
}
```
